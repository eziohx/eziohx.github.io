在Unity的text中实现打字机效果

以前的存货，能否使用未知。
应该不是最佳解决方法，但可以一用。

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;
public class text1 : MonoBehaviour{
  float time=0f;
  bool tf;
  Text Texta;
  string a="abcdefg";
  int i;
  void Awake(){
      Texta = GetComponent<Text> ();//如果不喜欢List可以用数组。
      tf=true;
  }
  void Update()
  {
      time+=Time.deltaTime;
      if(time>=0.2&&tf==true){
          Texta.text=a.Substring (0,i);
          i++;
          time=0f;
      }
      if(i>=a.Length){
          tf=false;
      }
  }
}
